GO 管道 channel

什么是管道

管道是Go语言在语言级别上提供的goroutine间的通讯方式，我们可以使用channel在多个goroutine之间传递消息。channel是进程内的通讯方式，是不支持跨进程通信的，如果需要进程间通讯的话，可以使用Socket等网络方式。

也可以简单的理解为：

在多个线程之间传递数据，同步数据的一种手段，而对通道的操作本身也是同步数据，同一时刻，仅有一个线程可以向 channel 发送数据，同样的，同一时刻也只能有一个线程能从 channel 读取数据。

chnnel 语法

能够存储数据的类型

chan T //T 为类型
chan int //缓冲int类型
chan float32 //缓冲float 类型
  • 可以设置别名

type IntChan chan int

声明

chan是声明管道的关键字

var a int //声明一个int类型变量a
var inChan chan int //只能缓冲int 类型
var strChan chan string //声明一个strChan管道，并可以写入string类型
//除了上面的声明方法还可以用`make`来声明
a:=make(chan int)
a:=make(chan int,10)

非缓冲chanel与缓冲channel对比

//非缓冲管道，必须等待通道取数据，会出现堵塞
a:=make(chan int)


//缓冲管道
a:=make(chan int,10)

len()//长度 上面初始化的非缓冲通道，缓冲通道都为0
cap()//容量  上面初始化的非缓冲通道为0，缓冲通道为10

非缓冲channel：

只能缓冲1个数据,无缓冲的读写实际为同步操作，必须一写一读操作匹配时，才会继续，否则会死锁。 非缓冲的读和写操作必须“相遇”才能执行，否则会阻塞。 比如快递员给你送快递，快递员送到你手里，签收完成后，快递员才可以离开

缓冲channel： 最大缓冲等于容量，相当于一个先进先出的队列。有缓冲的管道，只有当管道容量已存满，在未取走值的情况下继续放值时才会发生阻塞。

比如，快递员给你送快递，快递员把快递放到快递箱里面，如果快递箱放满了，没有取走情况下会出现堵塞

从管道中 取值 的方法以及一些案例

fmt.Println(<-c)//从管道中取值，取出c,并打印。

<-c //从管道中取值，不输出结果。(对于非缓存管道，必须把值取出来，才能放别的)

a := <-c//将管道中取出的值，赋值给 a

k, ok := <-c //从管道里面取值赋值给k，如果赋值成功，ok 为 true，如果赋值不成功，ok 为 flase

package main

import "fmt"

func main() {
	// 创建一个管道
	c := make(chan int, 1)
	// 查看写入管道中数据的数量
	fmt.Println(len(c))
	// 查看管道的缓冲容量
	fmt.Println(cap(c))

	c <- 10 //写入数据，这条管道只能写入int类型

	fmt.Println(len(c))
	fmt.Println(cap(c))

	// <-c //从管道里面取值
	// fmt.Println("从管道里面取值：", <-c)
	a := <-c //将管道里面取出来的值赋给a
	fmt.Println("a = ", a)
	c <- 20
	k, ok := <-c //从管理里面取值赋给k，如果赋值成功，ok为true

	fmt.Println(k, "  ", ok)
	fmt.Println(len(c))
	fmt.Println(cap(c))

}
输出为：
0
1
1
1
a =  10
20    true

下面案例是错的，造成死锁原因，没有复制，但一直等待取值

package main

import "fmt"

func main() {

	// 创建一个管道
	c := make(chan int, 1)

	// 从管道里面取值赋给a
	a, ok := <-c //管道内没有数据，无法取出，一直等待，出现死锁。

	fmt.Println(a, "  ", ok)

}

输出为
fatal error: all goroutines are asleep - deadlock!

goroutine 1 [chan receive]:
main.main()
	/BlockChain-learn/002_管道.go:11 +0x55
exit status 2
原因：//a, ok := <-c //管道内没有数据，无法取出，一直等待，出现死锁。

解决上面案例的方法

package main

import "fmt"

func main() {

	// 创建一个管道
	c := make(chan int, 1)

	c <- 10 //给管道传值

	// 从管道里面取值赋给a
	a, ok := <-c //同步

	fmt.Println(a, "  ", ok)

}
输出为
10    true

在子线程中去管道的值，让主线程休眠，给主线程足够读多时间

package main

import (
	"fmt"
	"time"
)

func main() {

	// 创建一个管道
	c := make(chan int, 1)

	go func() {
		// 从管道里面取值赋给a
		a, ok := <-c //同步
		fmt.Println(a, "  ", ok)
	}()

	c <- 10
	// 休眠一会儿是为了等待子线程处理任务
	time.Sleep(time.Second)

}
输出为
10    true

使用sync

package main

import (
	"fmt"
	"sync"
)

var sy sync.WaitGroup

func main() {

	// 创建一个管道
	c := make(chan int, 1)
	sy.Add(1)
	go func() { //子线程
		// 从管道里面取值赋给a
		a, ok := <-c //同步
		fmt.Println(a, "  ", ok)
		sy.Done()
	}()

	c <- 10
	sy.Wait()
}

输出为
10    true

channel 解决并发编程造成的死锁

下面channle与sync多线程编程的对比

非缓冲管道中，依次循环存入值，并逐个取出，并且在子线程

package main

import (
	"fmt"
	"sync"
)

var wg sync.WaitGroup //队列

func main() {

	// 先创建一个非缓冲管道
	c := make(chan int)
	wg.Add(2)
	go func() { //子线程
		for i := 0; i < 10; i++ {
			c <- i //写入数据
			fmt.Println("写入数据")
		}
		wg.Done()
	}()

	go func() {
		for i := 0; i < 10; i++ {
			a := <-c
			fmt.Println(a) //读取数据
			fmt.Println("读取数据")
		}
		wg.Done()
	}()

	wg.Wait()

}

用channel 实现，注意close与range结合


package main

import "fmt"

func main() {
	c := make(chan int)

	go func() {
		for i := 0; i < 10; i++ {
			fmt.Println("------")
			c <- i
		}
		close(c)
	}()

	// <-c

	for n := range c {
		fmt.Println("++++++++++")
		fmt.Println(n)
	}


}

注： for i := range c能够不断的读取channel里面的数据，直到该channel被显式的关闭。上面代码我们看到可以显式的关闭channel，生产者通过内置函数close关闭channel。关闭channel之后就无法再发送任何数据了，在消费方可以通过语法v, ok := <-ch测试channel是否被关闭。如果ok返回false，那么说明channel已经没有任何数据并且已经被关闭。

队列，多个子线程执行完毕后关闭再 管道：

package main

import (
	"fmt"
	"sync"
)

func main() {

	c := make(chan int) //创建通道

	var wg sync.WaitGroup //队列

	go func() {
		fmt.Println("111111111")
		wg.Add(1)
		for i := 0; i < 10; i++ {
			c <- i
		}
		wg.Done()
	}()

	go func() {
		fmt.Println("22222222")
		wg.Add(1)
		for i := 0; i < 10; i++ {
			c <- i
		}
		wg.Done()
	}()

	go func() {
		fmt.Println("33333333")
		wg.Wait()
		close(c)
	}()

	for n := range c {

		// <-c  有返回值
		fmt.Println(n)
	}
}

使用channel 多个子线程执行结束后，关闭通道：

package main

import (
	"fmt"
)

func main() {
//通过通道代替队列
	c := make(chan int)//
	done := make(chan bool)

	go func() {
		for i := 0; i < 10; i++ {
			c <- i
		}
		done <- true
	}()

	go func() {
		for i := 0; i < 10; i++ {
			c <- i
		}
		done <- true
	}()

	go func() {
		<-done//读取数据，如果通道done里面没有数据，就会出现线程堵塞，
		//直到有数据，才继续执行.
		//不在子线程 函数内，会死锁
		<-done
		close(c)
	}()

	for n := range c {
		fmt.Println(n)
	}
}
